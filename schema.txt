CREATE TABLE if not exists page_load (id integer primary key autoincrement, unique_code text not null unique, task_id integer,
                             ip text, host text, sent_time text, submit_time text,
                             expiration_time integer, create_reward real, vote_reward real,
                             items_create integer, items_rate integer, bonus_create real, bonus_vote real);

CREATE TABLE "feedback" ("id" INTEGER PRIMARY KEY  AUTOINCREMENT  NOT NULL , "text" TEXT NOT NULL , "worker_id" TEXT NOT NULL ,
 "task_id" INTEGER NOT NULL , "edit_time" TEXT);

CREATE TABLE if not exists task (id integer primary key autoincrement, text text,
                    requester_id text, requirement text, tasktype integer,
                    reward integer,  items_create integer, items_rate integer, 
                    target integer, expiration_time integer default 48, start_time text);
                    
CREATE TABLE if not exists message (id integer primary key autoincrement,
                            text text, worker_id text, edit_time text,
                            rating integer default 0,
                             task_id integer, valid integer default 1);               
                             
CREATE TABLE if not exists candidate (id integer primary key autoincrement,
                            message_id integer, worker_id integer, edit_time text);   
                            
CREATE TABLE if not exists pending_record (id integer primary key autoincrement,
                             worker_id text, mess_id integer,
                             task_id integer, edit_time text, valid integer default 1); 
                             
CREATE TABLE if not exists rating_record (id integer primary key autoincrement,
                             mess_id integer, temp_rating integer,
                             task_id integer, edit_time text);
